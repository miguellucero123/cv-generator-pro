/**
 * METGO_3D VIRTUALIZE - PDF Generator Composable

import { ref } from 'vue'

export function usePdfGenerator() {
  const isGenerating = ref(false)
  const error = ref(null)
  
  /**
   * Generate PDF from HTML element
   * @param {HTMLElement} element - DOM element to convert
   * @param {Object} options - Configuration options
   */
  const generatePDF = async (element, options = {}) => {
    isGenerating.value = true
    error.value = null
    
    const defaultOptions = {
      filename: 'CV_METGO3D.pdf',
      margin: [10, 10, 10, 10],
      image: { type: 'jpeg', quality: 0.98 },
      html2canvas: {
        scale: 2,
        useCORS: true,
        letterRendering: true,
        logging: false,
        backgroundColor: '#ffffff'
      },
      jsPDF: {
        unit: 'mm',
        format: 'a4',
        orientation: 'portrait',
        compress: true
      },
      pagebreak: { 
        mode: ['avoid-all', 'css', 'legacy'],
        before: '.page-break-before',
        after: '.page-break-after',
        avoid: '.page-break-avoid'
      }
    }
    
    const mergedOptions = {
      ...defaultOptions,
      ...options,
      html2canvas: { ...defaultOptions.html2canvas, ...options.html2canvas },
      jsPDF: { ...defaultOptions.jsPDF, ...options.jsPDF }
    }
    
    try {
      // Dynamic import for code splitting
      const html2pdf = (await import('html2pdf.js')).default
      
      // Clone element for PDF generation (to avoid modifying the original)
      const clonedElement = element.cloneNode(true)
      
      // Apply print-specific styles
      clonedElement.style.width = '210mm'
      clonedElement.style.padding = '10mm'
      clonedElement.style.backgroundColor = '#ffffff'
      
      // Generate PDF
      await html2pdf()
        .set(mergedOptions)
        .from(element)
        .save()
      
      // Analytics tracking (optional)
      if (typeof gtag !== 'undefined') {
        gtag('event', 'download_pdf', {
          event_category: 'CV',
          event_label: options.filename
        })
      }
      
      return true
    } catch (err) {
      console.error('Error generating PDF:', err)
      error.value = err.message || 'Error al generar el PDF'
      
      // Show user-friendly error
      alert(`Error al generar el PDF: ${error.value}\n\nPor favor, intente de nuevo o use la opci√≥n de imprimir.`)
      
      return false
    } finally {
      isGenerating.value = false
    }
  }
  
  /**
   * Generate PDF with custom styling
   */
  const generateStyledPDF = async (element, options = {}) => {
    // Add temporary print styles
    const styleElement = document.createElement('style')
    styleElement.id = 'pdf-temp-styles'
    styleElement.textContent = `
      .cv-preview {
        font-size: 11pt !important;
      }
      .skill-bar {
        display: none !important;
      }
      .animate-glow,
      .animate-float {
        animation: none !important;
      }
    `
    document.head.appendChild(styleElement)
    
    try {
      const result = await generatePDF(element, options)
      return result
    } finally {
      // Remove temporary styles
      document.getElementById('pdf-temp-styles')?.remove()
    }
  }
  
  return {
    isGenerating,
    error,
    generatePDF,
    generateStyledPDF
  }
}

/**
 * METGO_3D VIRTUALIZE - Theme Composable
 */

import { ref, watch } from 'vue'

const THEME_KEY = 'metgo3d-cv-theme'
const THEMES = ['light', 'dark', 'metgo']

export function useTheme() {
  const theme = ref('light')
  
  /**
   * Initialize theme from localStorage or system preference
   */
  const initTheme = () => {
    const savedTheme = localStorage.getItem(THEME_KEY)
    
    if (savedTheme && THEMES.includes(savedTheme)) {
      theme.value = savedTheme
    } else if (window.matchMedia?.('(prefers-color-scheme: dark)').matches) {
      theme.value = 'dark'
    }
    
    applyTheme(theme.value)
  }
  
  /**
   * Apply theme to document
   */
  const applyTheme = (themeName) => {
    document.documentElement.setAttribute('data-theme', themeName)
    
    // Update meta theme-color
    const metaThemeColor = document.querySelector('meta[name="theme-color"]')
    const colors = {
      light: '#f8fafc',
      dark: '#1a1a2e',
      metgo: '#0d0d1a'
    }
    
    if (metaThemeColor) {
      metaThemeColor.setAttribute('content', colors[themeName] || colors.light)
    }
  }
  
  /**
   * Toggle between themes
   */
  const toggleTheme = () => {
    const currentIndex = THEMES.indexOf(theme.value)
    const nextIndex = (currentIndex + 1) % THEMES.length
    theme.value = THEMES[nextIndex]
  }
  
  /**
   * Set specific theme
   */
  const setTheme = (themeName) => {
    if (THEMES.includes(themeName)) {
      theme.value = themeName
    }
  }
  
  // Watch for theme changes
  watch(theme, (newTheme) => {
    localStorage.setItem(THEME_KEY, newTheme)
    applyTheme(newTheme)
  })
  
  // Listen for system theme changes
  if (typeof window !== 'undefined') {
    window.matchMedia?.('(prefers-color-scheme: dark)')
      .addEventListener('change', (e) => {
        if (!localStorage.getItem(THEME_KEY)) {
          theme.value = e.matches ? 'dark' : 'light'
        }
      })
  }
  
  return {
    theme,
    themes: THEMES,
    initTheme,
    toggleTheme,
    setTheme
  }
}

/**
 * Format date to locale string
 * @param {Date|string} date 
 * @param {string} locale 
 * @returns {string}
 */
export function formatDate(date, locale = 'es-CL') {
  const options = { 
    year: 'numeric', 
    month: 'long', 
    day: 'numeric' 
  }
  return new Date(date).toLocaleDateString(locale, options)
}

/**
 * Format date to short format
 * @param {Date|string} date 
 * @returns {string}
 */
export function formatDateShort(date) {
  const options = { 
    year: 'numeric', 
    month: 'short'
  }
  return new Date(date).toLocaleDateString('es-CL', options)
}

/**
 * Format phone number
 * @param {string} phone 
 * @returns {string}
 */
export function formatPhone(phone) {
  const cleaned = phone.replace(/\D/g, '')
  if (cleaned.length === 11 && cleaned.startsWith('56')) {
    return `+${cleaned.slice(0, 2)} ${cleaned.slice(2, 3)} ${cleaned.slice(3, 7)} ${cleaned.slice(7)}`
  }
  return phone
}

/**
 * Truncate text with ellipsis
 * @param {string} text 
 * @param {number} length 
 * @returns {string}
 */
export function truncate(text, length = 100) {
  if (text.length <= length) return text
  return text.slice(0, length).trim() + '...'
}

/**
 * Capitalize first letter
 * @param {string} text 
 * @returns {string}
 */
export function capitalize(text) {
  return text.charAt(0).toUpperCase() + text.slice(1).toLowerCase()
}

/**
 * Format skill level to text
 * @param {number} level 
 * @returns {string}
 */
export function formatSkillLevel(level) {
  if (level >= 90) return 'Experto'
  if (level >= 75) return 'Avanzado'
  if (level >= 50) return 'Intermedio'
  if (level >= 25) return 'B√°sico'
  return 'Principiante'
}

/**
 * Generate initials from name
 * @param {string} name 
 * @returns {string}
 */
export function getInitials(name) {
  return name
    .split(' ')
    .map(word => word.charAt(0))
    .join('')
    .toUpperCase()
    .slice(0, 2)
}

/**
 * Format file size
 * @param {number} bytes 
 * @returns {string}
 */
export function formatFileSize(bytes) {
  if (bytes === 0) return '0 Bytes'
  const k = 1024
  const sizes = ['Bytes', 'KB', 'MB', 'GB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))
  return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i]
}
