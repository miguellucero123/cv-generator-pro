 * METGO_3D VIRTUALIZE - Editor Composable

import { ref, reactive, computed, watch } from 'vue'
import { useLocalStorage } from './useLocalStorage'
import { cvData as defaultCvData } from '../data/cvData'

// Clave para localStorage
const STORAGE_KEY = 'metgo3d-cv-data'

export function useEditor() {
  // Estado
  const { storedValue, setValue } = useLocalStorage(STORAGE_KEY, null)
  
  const cvData = reactive(storedValue.value ? JSON.parse(JSON.stringify(storedValue.value)) : JSON.parse(JSON.stringify(defaultCvData)))
  const isEditing = ref(false)
  const activeTab = ref('personal')
  const hasChanges = ref(false)
  const isSaving = ref(false)
  const lastSaved = ref(null)
  
  // Historial para undo/redo
  const history = ref([])
  const historyIndex = ref(-1)
  const maxHistorySize = 50
  
  // Tabs disponibles
  const tabs = [
    { id: 'personal', label: 'InformaciÃ³n Personal', icon: 'fas fa-user' },
    { id: 'profile', label: 'Perfil', icon: 'fas fa-id-card' },
    { id: 'skills', label: 'Competencias', icon: 'fas fa-cogs' },
    { id: 'education', label: 'EducaciÃ³n', icon: 'fas fa-graduation-cap' },
    { id: 'experience', label: 'Experiencia', icon: 'fas fa-briefcase' },
    { id: 'projects', label: 'Proyectos', icon: 'fas fa-rocket' },
    { id: 'additional', label: 'Adicional', icon: 'fas fa-info-circle' }
  ]
  
  // Computed
  const canUndo = computed(() => historyIndex.value > 0)
  const canRedo = computed(() => historyIndex.value < history.value.length - 1)
  const completionPercentage = computed(() => calculateCompletion())
  
  // MÃ©todos
  const toggleEditor = () => {
    isEditing.value = !isEditing.value
  }
  
  const setActiveTab = (tabId) => {
    activeTab.value = tabId
  }
  
  const updateField = (section, field, value) => {
    // Guardar estado actual en historial
    saveToHistory()
    
    // Actualizar campo
    if (field.includes('.')) {
      const keys = field.split('.')
      let target = cvData[section]
      for (let i = 0; i < keys.length - 1; i++) {
        target = target[keys[i]]
      }
      target[keys[keys.length - 1]] = value
    } else {
      cvData[section][field] = value
    }
    
    hasChanges.value = true
  }
  
  const updateArrayItem = (section, index, field, value) => {
    saveToHistory()
    
    if (field) {
      cvData[section][index][field] = value
    } else {
      cvData[section][index] = value
    }
    
    hasChanges.value = true
  }
  
  const addArrayItem = (section, item) => {
    saveToHistory()
    cvData[section].push(item)
    hasChanges.value = true
  }
  
  const removeArrayItem = (section, index) => {
    saveToHistory()
    cvData[section].splice(index, 1)
    hasChanges.value = true
  }
  
  const moveArrayItem = (section, fromIndex, toIndex) => {
    saveToHistory()
    const item = cvData[section].splice(fromIndex, 1)[0]
    cvData[section].splice(toIndex, 0, item)
    hasChanges.value = true
  }
  
  const saveToHistory = () => {
    // Eliminar estados futuros si estamos en medio del historial
    if (historyIndex.value < history.value.length - 1) {
      history.value = history.value.slice(0, historyIndex.value + 1)
    }
    
    // Agregar estado actual
    history.value.push(JSON.stringify(cvData))
    
    // Limitar tamaÃ±o del historial
    if (history.value.length > maxHistorySize) {
      history.value.shift()
    } else {
      historyIndex.value++
    }
  }
  
  const undo = () => {
    if (canUndo.value) {
      historyIndex.value--
      const previousState = JSON.parse(history.value[historyIndex.value])
      Object.assign(cvData, previousState)
      hasChanges.value = true
    }
  }
  
  const redo = () => {
    if (canRedo.value) {
      historyIndex.value++
      const nextState = JSON.parse(history.value[historyIndex.value])
      Object.assign(cvData, nextState)
      hasChanges.value = true
    }
  }
  
  const saveChanges = async () => {
    isSaving.value = true
    
    try {
      // Simular delay de guardado
      await new Promise(resolve => setTimeout(resolve, 500))
      
      // Guardar en localStorage
      setValue(JSON.parse(JSON.stringify(cvData)))
      
      hasChanges.value = false
      lastSaved.value = new Date()
      
      return true
    } catch (error) {
      console.error('Error saving changes:', error)
      return false
    } finally {
      isSaving.value = false
    }
  }
  
  const resetToDefault = () => {
    if (confirm('Â¿EstÃ¡s seguro de que deseas restaurar los datos por defecto? Se perderÃ¡n todos los cambios.')) {
      saveToHistory()
      Object.assign(cvData, JSON.parse(JSON.stringify(defaultCvData)))
      hasChanges.value = true
    }
  }
  
  const exportData = () => {
    const dataStr = JSON.stringify(cvData, null, 2)
    const dataBlob = new Blob([dataStr], { type: 'application/json' })
    const url = URL.createObjectURL(dataBlob)
    
    const link = document.createElement('a')
    link.href = url
    link.download = `cv-data-${new Date().toISOString().split('T')[0]}.json`
    link.click()
    
    URL.revokeObjectURL(url)
  }
  
  const importData = (file) => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader()
      
      reader.onload = (e) => {
        try {
          const importedData = JSON.parse(e.target.result)
          saveToHistory()
          Object.assign(cvData, importedData)
          hasChanges.value = true
          resolve(true)
        } catch (error) {
          reject(new Error('Archivo JSON invÃ¡lido'))
        }
      }
      
      reader.onerror = () => reject(new Error('Error al leer el archivo'))
      reader.readAsText(file)
    })
  }
  
  const calculateCompletion = () => {
    let completed = 0
    let total = 0
    
    // Personal (8 campos)
    const personalFields = ['nombre', 'titulo', 'email', 'telefono', 'ubicacion', 'linkedin', 'github', 'website']
    personalFields.forEach(field => {
      total++
      if (cvData.personal[field]) completed++
    })
    
    // Perfil (2 campos)
    total += 2
    if (cvData.perfil.resumen) completed++
    if (cvData.perfil.propuestaValor) completed++
    
    // Competencias (4 categorÃ­as)
    total += 4
    if (cvData.competencias.programacion?.skills?.length > 0) completed++
    if (cvData.competencias.datascience?.skills?.length > 0) completed++
    if (cvData.competencias.herramientas?.skills?.length > 0) completed++
    if (cvData.competencias.soft?.skills?.length > 0) completed++
    
    // EducaciÃ³n (mÃ­nimo 1)
    total++
    if (cvData.educacion?.length > 0) completed++
    
    // Experiencia (mÃ­nimo 1)
    total++
    if (cvData.experiencia?.length > 0) completed++
    
    // Proyectos (mÃ­nimo 1)
    total++
    if (cvData.proyectos?.length > 0) completed++
    
    return Math.round((completed / total) * 100)
  }
  
  // Guardar automÃ¡ticamente cuando hay cambios
  watch(hasChanges, (newValue) => {
    if (newValue) {
      // Auto-save despuÃ©s de 3 segundos de inactividad
      const timeoutId = setTimeout(() => {
        saveChanges()
      }, 3000)
      
      return () => clearTimeout(timeoutId)
    }
  })
  
  // Inicializar historial
  history.value.push(JSON.stringify(cvData))
  historyIndex.value = 0
  
  return {
    // Estado
    cvData,
    isEditing,
    activeTab,
    hasChanges,
    isSaving,
    lastSaved,
    tabs,
    
    // Computed
    canUndo,
    canRedo,
    completionPercentage,
    
    // MÃ©todos
    toggleEditor,
    setActiveTab,
    updateField,
    updateArrayItem,
    addArrayItem,
    removeArrayItem,
    moveArrayItem,
    undo,
    redo,
    saveChanges,
    resetToDefault,
    exportData,
    importData
  }
}

/**
 * METGO_3D VIRTUALIZE - LocalStorage Composable
 */

import { ref, watch } from 'vue'

export function useLocalStorage(key, defaultValue) {
  // Obtener valor inicial
  const getStoredValue = () => {
    try {
      const item = localStorage.getItem(key)
      return item ? JSON.parse(item) : defaultValue
    } catch (error) {
      console.error('Error reading from localStorage:', error)
      return defaultValue
    }
  }
  
  const storedValue = ref(getStoredValue())
  
  // Guardar valor
  const setValue = (value) => {
    try {
      storedValue.value = value
      localStorage.setItem(key, JSON.stringify(value))
    } catch (error) {
      console.error('Error writing to localStorage:', error)
    }
  }
  
  // Eliminar valor
  const removeValue = () => {
    try {
      storedValue.value = defaultValue
      localStorage.removeItem(key)
    } catch (error) {
      console.error('Error removing from localStorage:', error)
    }
  }
  
  // Watch para sincronizar
  watch(storedValue, (newValue) => {
    localStorage.setItem(key, JSON.stringify(newValue))
  }, { deep: true })
  
  // Escuchar cambios de otras pestaÃ±as
  if (typeof window !== 'undefined') {
    window.addEventListener('storage', (e) => {
      if (e.key === key) {
        storedValue.value = e.newValue ? JSON.parse(e.newValue) : defaultValue
      }
    })
  }
  
  return {
    storedValue,
    setValue,
    removeValue
  }
}

    }
    
    const currentFormComponent = computed(() => {
      return formComponents[props.activeTab] || 'PersonalForm'
    })
    
    const completionClass = computed(() => {
      const p = props.completionPercentage
      if (p >= 80) return 'high'
      if (p >= 50) return 'medium'
      return 'low'
    })
    
    const formatTime = (date) => {
      if (!date) return ''
      const now = new Date()
      const diff = now - date
      
      if (diff < 60000) return 'hace un momento'
      if (diff < 3600000) return `hace ${Math.floor(diff / 60000)} minutos`
      return date.toLocaleTimeString('es-CL', { hour: '2-digit', minute: '2-digit' })
    }
    
    const setActiveTab = (tabId) => emit('set-tab', tabId)
    const undo = () => emit('undo')
    const redo = () => emit('redo')
    const resetToDefault = () => emit('reset')
    const handleExport = () => emit('export')
    
    const handleImport = (event) => {
      const file = event.target.files[0]
      if (file) {
        emit('import', file)
        event.target.value = ''
      }
    }
    
    const handleSave = () => emit('save')
    
    const handleUpdate = (section, field, value) => {
      emit('update', section, field, value)
    }
    
    const handleUpdateArray = (section, index, field, value) => {
      emit('update-array', section, index, field, value)
    }
    
    const handleAdd = (section, item) => {
      emit('add', section, item)
    }
    
    const handleRemove = (section, index) => {
      emit('remove', section, index)
    }
    
    const handleMove = (section, fromIndex, toIndex) => {
      emit('move', section, fromIndex, toIndex)
    }
    
    return {
      currentFormComponent,
      completionClass,
      formatTime,
      setActiveTab,
      undo,
      redo,
      resetToDefault,
      handleExport,
      handleImport,
      handleSave,
      handleUpdate,
      handleUpdateArray,
      handleAdd,
      handleRemove,
      handleMove
    }
  }
}

      emit('update', 'competencias', 'soft.skills', skills)
    }
    
    const addSoftSkill = () => {
      const skills = [...props.data.competencias.soft.skills, '']
      emit('update', 'competencias', 'soft.skills', skills)
    }
    
    const removeSoftSkill = (index) => {
      const skills = props.data.competencias.soft.skills.filter((_, i) => i !== index)
      emit('update', 'competencias', 'soft.skills', skills)
    }
    
    return {
      updateSkill,
      addSkill,
      removeSkill,
      updateSoftSkill,
      addSoftSkill,
      removeSoftSkill
    }

import { ref, reactive } from 'vue'
import FormField from './FormField.vue'

export default {
  name: 'ExperienceForm',
  
  components: {
    FormField
  },
  
  props: {
    data: {
      type: Object,
      required: true
    }
  },
  
  emits: ['update', 'update-array', 'add', 'remove', 'move'],
  
  setup(props, { emit }) {
    const expandedIndex = ref(0)
    const newTech = reactive({})
    
    const tipoOptions = [
      { value: 'Tiempo completo', label: 'Tiempo completo' },
      { value: 'Medio tiempo', label: 'Medio tiempo' },
      { value: 'Freelance', label: 'Freelance' },
      { value: 'ConsultorÃ­a', label: 'ConsultorÃ­a' },
      { value: 'PrÃ¡ctica', label: 'PrÃ¡ctica profesional' },
      { value: 'Contrato', label: 'Contrato temporal' }
    ]
    
    const toggleExpand = (index) => {
      expandedIndex.value = expandedIndex.value === index ? -1 : index
    }
    
    const updateExp = (index, field, value) => {
      emit('update-array', 'experiencia', index, field, value)
    }
    
    const updateLogro = (expIndex, logroIndex, field, value) => {
      const logros = [...props.data.experiencia[expIndex].logros]
      logros[logroIndex] = { ...logros[logroIndex], [field]: value }
      emit('update-array', 'experiencia', expIndex, 'logros', logros)
    }
    
    const addLogro = (expIndex) => {
      const logros = [
        ...props.data.experiencia[expIndex].logros,
        { descripcion: '', metrica: null, impacto: '' }
      ]
      emit('update-array', 'experiencia', expIndex, 'logros', logros)
    }
    
    const removeLogro = (expIndex, logroIndex) => {
      const logros = props.data.experiencia[expIndex].logros.filter((_, i) => i !== logroIndex)
      emit('update-array', 'experiencia', expIndex, 'logros', logros)
    }
    
    const addTech = (expIndex) => {
      if (!newTech[expIndex]?.trim()) return
      
      const tecnologias = [
        ...(props.data.experiencia[expIndex].tecnologias || []),
        newTech[expIndex].trim()
      ]
      emit('update-array', 'experiencia', expIndex, 'tecnologias', tecnologias)
      newTech[expIndex] = ''
    }
    
    const removeTech = (expIndex, techIndex) => {
      const tecnologias = props.data.experiencia[expIndex].tecnologias.filter((_, i) => i !== techIndex)
      emit('update-array', 'experiencia', expIndex, 'tecnologias', tecnologias)
    }
    
    const addExperience = () => {
      const newExp = {
        cargo: '',
        empresa: '',
        proyecto: null,
        ubicacion: '',
        periodo: '',
        duracion: '',
        tipo: 'Tiempo completo',
        industria: '',
        logros: [{ descripcion: '', metrica: null, impacto: '' }],
        tecnologias: [],
        destacado: false
      }
      emit('add', 'experiencia', newExp)
      expandedIndex.value = props.data.experiencia.length
    }
    
    const removeExperience = (index) => {
      if (confirm('Â¿EstÃ¡s seguro de eliminar esta experiencia?')) {
        emit('remove', 'experiencia', index)
        if (expandedIndex.value === index) {
          expandedIndex.value = -1
        }
      }
    }
    
    const moveUp = (index) => {
      if (index > 0) {
        emit('move', 'experiencia', index, index - 1)
        expandedIndex.value = index - 1
      }
    }
    
    const moveDown = (index) => {
      if (index < props.data.experiencia.length - 1) {
        emit('move', 'experiencia', index, index + 1)
        expandedIndex.value = index + 1
      }
    }
    
    return {
      expandedIndex,
      newTech,
      tipoOptions,
      toggleExpand,
      updateExp,
      updateLogro,
      addLogro,
      removeLogro,
      addTech,
      removeTech,
      addExperience,
      removeExperience,
      moveUp,
      moveDown
    }
  }
}

export function useTemplates() {
  const currentTemplate = ref('modern')
  const isChanging = ref(false)
  
  const templateList = computed(() => Object.values(templates))
  
  const activeTemplate = computed(() => templates[currentTemplate.value])
  
  const setTemplate = async (templateId) => {
    if (!templates[templateId] || templateId === currentTemplate.value) return
    
    isChanging.value = true
    
    // Simulate transition delay
    await new Promise(resolve => setTimeout(resolve, 300))
    
    currentTemplate.value = templateId
    
    // Apply template-specific CSS variables
    applyTemplateStyles(templateId)
    
    // Save preference
    localStorage.setItem('metgo3d-cv-template', templateId)
    
    isChanging.value = false
  }
  
  const applyTemplateStyles = (templateId) => {
    const template = templates[templateId]
    if (!template) return
    
    const root = document.documentElement
    root.style.setProperty('--template-primary', template.colors.primary)
    root.style.setProperty('--template-secondary', template.colors.secondary)
    root.style.setProperty('--template-accent', template.colors.accent)
  }
  
  const initTemplate = () => {
    const saved = localStorage.getItem('metgo3d-cv-template')
    if (saved && templates[saved]) {
      currentTemplate.value = saved
      applyTemplateStyles(saved)
    }
  }
  
  return {
    currentTemplate,
    activeTemplate,
    templateList,
    isChanging,
    setTemplate,
    initTemplate
  }
}

/**
 * METGO_3D VIRTUALIZE - QR Code Composable
 */

import { ref, computed } from 'vue'

export function useQRCode() {
  const isGenerating = ref(false)
  const error = ref(null)
  
  /**
   * Generate QR Code as Data URL
   * @param {string} data - Data to encode
   * @param {Object} options - QR options
   * @returns {Promise<string>} Data URL
   */
  const generateQR = async (data, options = {}) => {
    isGenerating.value = true
    error.value = null
    
    const defaultOptions = {
      width: 200,
      height: 200,
      margin: 2,
      color: {
        dark: '#000000',
        light: '#ffffff'
      },
      errorCorrectionLevel: 'M'
    }
    
    const mergedOptions = { ...defaultOptions, ...options }
    
    try {
      const QRCode = (await import('qrcode')).default
      
      const dataUrl = await QRCode.toDataURL(data, {
        width: mergedOptions.width,
        margin: mergedOptions.margin,
        color: mergedOptions.color,
        errorCorrectionLevel: mergedOptions.errorCorrectionLevel
      })
      
      return dataUrl
    } catch (err) {
      console.error('Error generating QR code:', err)
      error.value = err.message
      return null
    } finally {
      isGenerating.value = false
    }
  }
  
  /**
   * Generate QR Code to Canvas
   * @param {HTMLCanvasElement} canvas - Canvas element
   * @param {string} data - Data to encode
   * @param {Object} options - QR options
   */
  const generateToCanvas = async (canvas, data, options = {}) => {
    isGenerating.value = true
    error.value = null
    
    try {
      const QRCode = (await import('qrcode')).default
      
      await QRCode.toCanvas(canvas, data, {
        width: options.width || 200,
        margin: options.margin || 2,
        color: options.color || { dark: '#000000', light: '#ffffff' }
      })
      
      return true
    } catch (err) {
      console.error('Error generating QR to canvas:', err)
      error.value = err.message
      return false
    } finally {
      isGenerating.value = false
    }
  }
  
  /**
   * Generate vCard QR Code
   * @param {Object} contact - Contact information
   * @returns {Promise<string>} Data URL
   */
  const generateVCard = async (contact) => {
    const vCard = `BEGIN:VCARD
VERSION:3.0
FN:${contact.nombre}
TITLE:${contact.titulo || ''}
TEL:${contact.telefono || ''}
EMAIL:${contact.email || ''}
URL:${contact.website || ''}
ADR:;;${contact.ubicacion || ''}
END:VCARD`
    
    return generateQR(vCard)
  }
  
  return {
    isGenerating,
    error,
    generateQR,
    generateToCanvas,
    generateVCard
  }
}

/**
 * METGO_3D VIRTUALIZE - Scroll Animations Composable
 */

import { ref, onMounted, onUnmounted } from 'vue'

export function useScrollAnimations(options = {}) {
  const observedElements = ref(new Set())
  let observer = null
  
  const defaultOptions = {
    threshold: 0.1,
    rootMargin: '0px 0px -50px 0px',
    animationClass: 'is-visible',
    staggerDelay: 100
  }
  
  const config = { ...defaultOptions, ...options }
  
  /**
   * Initialize Intersection Observer
   */
  const initObserver = () => {
    if (typeof window === 'undefined' || !('IntersectionObserver' in window)) {
      return
    }
    
    observer = new IntersectionObserver((entries) => {
      entries.forEach((entry, index) => {
        if (entry.isIntersecting) {
          // Add stagger delay
          const delay = index * config.staggerDelay
          
          setTimeout(() => {
            entry.target.classList.add(config.animationClass)
          }, delay)
          
          // Unobserve after animation
          if (!config.repeat) {
            observer.unobserve(entry.target)
            observedElements.value.delete(entry.target)
          }
        } else if (config.repeat) {
          entry.target.classList.remove(config.animationClass)
        }
      })
    }, {
      threshold: config.threshold,
      rootMargin: config.rootMargin
    })
  }
  
  /**
   * Observe an element
   * @param {HTMLElement} element 
   */
  const observe = (element) => {
    if (!observer || !element) return
    
    observer.observe(element)
    observedElements.value.add(element)
  }
  
  /**
   * Observe multiple elements
   * @param {string} selector - CSS selector
   */
  const observeAll = (selector) => {
    const elements = document.querySelectorAll(selector)
    elements.forEach(el => observe(el))
  }
  
  /**
   * Unobserve an element
   * @param {HTMLElement} element 
   */
  const unobserve = (element) => {
    if (!observer || !element) return
    
    observer.unobserve(element)
    observedElements.value.delete(element)
  }
  
  /**
   * Unobserve all elements
   */
  const unobserveAll = () => {
    observedElements.value.forEach(el => {
      observer?.unobserve(el)
    })
    observedElements.value.clear()
  }
  
  /**
   * Disconnect observer
   */
  const disconnect = () => {
    observer?.disconnect()
    observedElements.value.clear()
  }
  
  /**
   * Vue directive for scroll animations
   */
  const vScrollAnimate = {
    mounted(el, binding) {
      const animationType = binding.value || 'fade-up'
      el.classList.add('scroll-animate', `animate-${animationType}`)
      observe(el)
    },
    unmounted(el) {
      unobserve(el)
    }
  }
  
  // Lifecycle
  onMounted(() => {
    initObserver()
  })
  
  onUnmounted(() => {
    disconnect()
  })
  
  return {
    observe,
    observeAll,
    unobserve,
    unobserveAll,
    disconnect,
    vScrollAnimate
  }
}

/**
 * METGO_3D VIRTUALIZE - Presentation Mode Composable
 */

import { ref, computed, onMounted, onUnmounted } from 'vue'

export function usePresentation(slides = []) {
  const isActive = ref(false)
  const currentSlide = ref(0)
  const isTransitioning = ref(false)
  const isPaused = ref(false)
  const autoPlayInterval = ref(null)
  const autoPlayDelay = ref(5000) // 5 seconds
  
  // Computed
  const totalSlides = computed(() => slides.length)
  const hasNext = computed(() => currentSlide.value < totalSlides.value - 1)
  const hasPrev = computed(() => currentSlide.value > 0)
  const progress = computed(() => ((currentSlide.value + 1) / totalSlides.value) * 100)
  const currentSlideData = computed(() => slides[currentSlide.value] || null)
  
  // Methods
  const startPresentation = () => {
    isActive.value = true
    currentSlide.value = 0
    document.body.style.overflow = 'hidden'
    
    // Enter fullscreen if supported
    if (document.documentElement.requestFullscreen) {
      document.documentElement.requestFullscreen().catch(() => {})
    }
  }
  
  const endPresentation = () => {
    isActive.value = false
    stopAutoPlay()
    document.body.style.overflow = ''
    
    // Exit fullscreen
    if (document.fullscreenElement) {
      document.exitFullscreen().catch(() => {})
    }
  }
  
  const nextSlide = () => {
    if (!hasNext.value || isTransitioning.value) return
    
    isTransitioning.value = true
    currentSlide.value++
    
    setTimeout(() => {
      isTransitioning.value = false
    }, 500)
  }
  
  const prevSlide = () => {
    if (!hasPrev.value || isTransitioning.value) return
    
    isTransitioning.value = true
    currentSlide.value--
    
    setTimeout(() => {
      isTransitioning.value = false
    }, 500)
  }
  
  const goToSlide = (index) => {
    if (index < 0 || index >= totalSlides.value || isTransitioning.value) return
    
    isTransitioning.value = true
    currentSlide.value = index
    
    setTimeout(() => {
      isTransitioning.value = false
    }, 500)
  }
  
  const startAutoPlay = () => {
    isPaused.value = false
    autoPlayInterval.value = setInterval(() => {
      if (hasNext.value) {
        nextSlide()
      } else {
        currentSlide.value = 0
      }
    }, autoPlayDelay.value)
  }
  
  const stopAutoPlay = () => {
    isPaused.value = true
    if (autoPlayInterval.value) {
      clearInterval(autoPlayInterval.value)
      autoPlayInterval.value = null
    }
  }
  
  const toggleAutoPlay = () => {
    if (isPaused.value) {
      startAutoPlay()
    } else {
      stopAutoPlay()
    }
  }
  
  // Keyboard navigation
  const handleKeydown = (e) => {
    if (!isActive.value) return
    
    switch (e.key) {
      case 'ArrowRight':
      case 'Space':
      case 'Enter':
        e.preventDefault()
        nextSlide()
        break
      case 'ArrowLeft':
      case 'Backspace':
        e.preventDefault()
        prevSlide()
        break
      case 'Escape':
        endPresentation()
        break
      case 'Home':
        goToSlide(0)
        break
      case 'End':
        goToSlide(totalSlides.value - 1)
        break
      case 'p':
      case 'P':
        toggleAutoPlay()
        break
    }
  }
  
  // Touch swipe support
  let touchStartX = 0
  let touchEndX = 0
  
  const handleTouchStart = (e) => {
    touchStartX = e.changedTouches[0].screenX
  }
  
  const handleTouchEnd = (e) => {
    touchEndX = e.changedTouches[0].screenX
    handleSwipe()
  }
  
  const handleSwipe = () => {
    const threshold = 50
    const diff = touchStartX - touchEndX
    
    if (Math.abs(diff) > threshold) {
      if (diff > 0) {
        nextSlide()
      } else {
        prevSlide()
      }
    }
  }
  
  // Lifecycle
  onMounted(() => {
    window.addEventListener('keydown', handleKeydown)
  })
  
  onUnmounted(() => {
    window.removeEventListener('keydown', handleKeydown)
    stopAutoPlay()
  })
  
  return {
    // State
    isActive,
    currentSlide,
    isTransitioning,
    isPaused,
    
    // Computed
    totalSlides,
    hasNext,
    hasPrev,
    progress,
    currentSlideData,
    
    // Methods
    startPresentation,
    endPresentation,
    nextSlide,
    prevSlide,
    goToSlide,
    startAutoPlay,
    stopAutoPlay,
    toggleAutoPlay,
    
    // Touch handlers
    handleTouchStart,
    handleTouchEnd
  }
}

<script>
import { computed } from 'vue'

export default {
  name: 'SlideView',
  
  props: {
    slide: {
      type: Object,
      required: true
    },
    data: {
      type: Object,
      default: null
    }
  },
  
  setup(props) {
    const initials = computed(() => {
      if (!props.data?.personal?.nombre) return '??'
      return props.data.personal.nombre
        .split(' ')
        .map(w => w[0])
        .join('')
        .slice(0, 2)
        .toUpperCase()
    })
    
    const skillCategories = computed(() => {
      if (!props.data?.competencias) return {}
      const { programacion, datascience, herramientas } = props.data.competencias
      return {
        programacion: { ...programacion, icon: 'fas fa-code' },
        datascience: { ...datascience, icon: 'fas fa-chart-line' },
        herramientas: { ...herramientas, icon: 'fas fa-tools' }
      }
    })
    
    return {
      initials,
      skillCategories
    }
  }
}
</script>

/**
 * METGO_3D VIRTUALIZE - Internationalization Composable
 */

import { ref, computed, reactive } from 'vue'
import es from '../i18n/es'
import en from '../i18n/en'

const STORAGE_KEY = 'metgo3d-locale'
const DEFAULT_LOCALE = 'es'

const locales = {
  es,
  en
}

const availableLocales = [
  { code: 'es', name: 'EspaÃ±ol', flag: 'ğŸ‡¨ğŸ‡±' },
  { code: 'en', name: 'English', flag: 'ğŸ‡ºğŸ‡¸' }
]

export function useI18n() {
  // State
  const currentLocale = ref(localStorage.getItem(STORAGE_KEY) || DEFAULT_LOCALE)
  const translations = reactive(locales[currentLocale.value] || locales[DEFAULT_LOCALE])
  
  // Computed
  const locale = computed(() => currentLocale.value)
  const localeInfo = computed(() => 
    availableLocales.find(l => l.code === currentLocale.value) || availableLocales[0]
  )
  
  /**
   * Translate a key
   * @param {string} key - Translation key (dot notation)
   * @param {Object} params - Interpolation parameters
   * @returns {string}
   */
  const t = (key, params = {}) => {
    const keys = key.split('.')
    let value = translations
    
    for (const k of keys) {
      if (value && typeof value === 'object' && k in value) {
        value = value[k]
      } else {
        console.warn(`Translation key not found: ${key}`)
        return key
      }
    }
    
    if (typeof value !== 'string') {
      return key
    }
    
    // Interpolation
    return value.replace(/\{(\w+)\}/g, (match, paramKey) => {
      return params[paramKey] !== undefined ? params[paramKey] : match
    })
  }
  
  /**
   * Set locale
   * @param {string} localeCode 
   */
  const setLocale = (localeCode) => {
    if (!locales[localeCode]) {
      console.warn(`Locale not available: ${localeCode}`)
      return
    }
    
    currentLocale.value = localeCode
    Object.assign(translations, locales[localeCode])
    localStorage.setItem(STORAGE_KEY, localeCode)
    
    // Update document lang
    document.documentElement.setAttribute('lang', localeCode)
  }
  
  /**
   * Toggle between available locales
   */
  const toggleLocale = () => {
    const currentIndex = availableLocales.findIndex(l => l.code === currentLocale.value)
    const nextIndex = (currentIndex + 1) % availableLocales.length
    setLocale(availableLocales[nextIndex].code)
  }
  
  /**
   * Check if a translation key exists
   * @param {string} key 
   * @returns {boolean}
   */
  const hasTranslation = (key) => {
    const keys = key.split('.')
    let value = translations
    
    for (const k of keys) {
      if (value && typeof value === 'object' && k in value) {
        value = value[k]
      } else {
        return false
      }
    }
    
    return typeof value === 'string'
  }
  
  /**
   * Get all translations for a namespace
   * @param {string} namespace 
   * @returns {Object}
   */
  const getNamespace = (namespace) => {
    const keys = namespace.split('.')
    let value = translations
    
    for (const k of keys) {
      if (value && typeof value === 'object' && k in value) {
        value = value[k]
      } else {
        return {}
      }
    }
    
    return typeof value === 'object' ? value : {}
  }
  
  // Initialize
  const init = () => {
    const savedLocale = localStorage.getItem(STORAGE_KEY)
    if (savedLocale && locales[savedLocale]) {
      setLocale(savedLocale)
    } else {
      // Detect browser language
      const browserLang = navigator.language.split('-')[0]
      if (locales[browserLang]) {
        setLocale(browserLang)
      }
    }
  }
  
  return {
    // State
    locale,
    localeInfo,
    availableLocales,
    
    // Methods
    t,
    setLocale,
    toggleLocale,
    hasTranslation,
    getNamespace,
    init
  }
}

/**
 * METGO_3D VIRTUALIZE - Toast Notifications Composable
 */

import { ref, reactive } from 'vue'

const toasts = reactive([])
let toastId = 0

export function useToast() {
  const defaultOptions = {
    duration: 4000,
    position: 'bottom-right',
    closable: true
  }
  
  /**
   * Show a toast notification
   * @param {Object} options - Toast options
   * @returns {number} Toast ID
   */
  const show = (options) => {
    const id = ++toastId
    
    const toast = {
      id,
      type: options.type || 'info',
      title: options.title || '',
      message: options.message || '',
      icon: options.icon || getDefaultIcon(options.type),
      duration: options.duration ?? defaultOptions.duration,
      position: options.position || defaultOptions.position,
      closable: options.closable ?? defaultOptions.closable,
      action: options.action || null,
      progress: options.duration > 0,
      createdAt: Date.now()
    }
    
    toasts.push(toast)
    
    // Auto remove
    if (toast.duration > 0) {
      setTimeout(() => {
        remove(id)
      }, toast.duration)
    }
    
    return id
  }
  
  /**
   * Show success toast
   */
  const success = (message, options = {}) => {
    return show({
      ...options,
      type: 'success',
      message,
      title: options.title || 'Ã‰xito'
    })
  }
  
  /**
   * Show error toast
   */
  const error = (message, options = {}) => {
    return show({
      ...options,
      type: 'error',
      message,
      title: options.title || 'Error',
      duration: options.duration ?? 6000
    })
  }
  
  /**
   * Show warning toast
   */
  const warning = (message, options = {}) => {
    return show({
      ...options,
      type: 'warning',
      message,
      title: options.title || 'Advertencia'
    })
  }
  
  /**
   * Show info toast
   */
  const info = (message, options = {}) => {
    return show({
      ...options,
      type: 'info',
      message,
      title: options.title || 'InformaciÃ³n'
    })
  }
  
  /**
   * Show loading toast
   */
  const loading = (message, options = {}) => {
    return show({
      ...options,
      type: 'loading',
      message,
      title: options.title || 'Cargando',
      duration: 0,
      closable: false
    })
  }
  
  /**
   * Update existing toast
   */
  const update = (id, options) => {
    const index = toasts.findIndex(t => t.id === id)
    if (index !== -1) {
      Object.assign(toasts[index], options)
      
      // If duration is set, schedule removal
      if (options.duration && options.duration > 0) {
        setTimeout(() => {
          remove(id)
        }, options.duration)
      }
    }
  }
  
  /**
   * Remove toast by ID
   */
  const remove = (id) => {
    const index = toasts.findIndex(t => t.id === id)
    if (index !== -1) {
      toasts.splice(index, 1)
    }
  }
  
  /**
   * Remove all toasts
   */
  const removeAll = () => {
    toasts.splice(0, toasts.length)
  }
  
  /**
   * Get default icon for toast type
   */
  const getDefaultIcon = (type) => {
    const icons = {
      success: 'fas fa-check-circle',
      error: 'fas fa-exclamation-circle',
      warning: 'fas fa-exclamation-triangle',
      info: 'fas fa-info-circle',
      loading: 'fas fa-spinner fa-spin'
    }
    return icons[type] || icons.info
  }
  
  /**
   * Promise-based toast for async operations
   */
  const promise = async (promiseFn, options = {}) => {
    const {
      loading: loadingMsg = 'Procesando...',
      success: successMsg = 'OperaciÃ³n completada',
      error: errorMsg = 'Ha ocurrido un error'
    } = options
    
    const id = loading(loadingMsg)
    
    try {
      const result = await promiseFn()
      update(id, {
        type: 'success',
        title: 'Ã‰xito',
        message: typeof successMsg === 'function' ? successMsg(result) : successMsg,
        icon: 'fas fa-check-circle',
        duration: 4000,
        closable: true
      })
      return result
    } catch (err) {
      update(id, {
        type: 'error',
        title: 'Error',
        message: typeof errorMsg === 'function' ? errorMsg(err) : errorMsg,
        icon: 'fas fa-exclamation-circle',
        duration: 6000,
        closable: true
      })
      throw err
    }
  }
  
  return {
    toasts,
    show,
    success,
    error,
    warning,
    info,
    loading,
    update,
    remove,
    removeAll,
    promise
  }
}

export function getToast() {
  if (!globalToast) {
    globalToast = useToast()
  }
  return globalToast
}

        'top-left',

        'bottom-center',
        'bottom-right'
      ].includes(value)

      type: Number,
      default: 5

  }

          <!-- Body -->

            <slot></slot>

<script>
import { watch, onMounted, onUnmounted } from 'vue'

export default {
  name: 'AppModal',
  
  props: {
    isOpen: {
      type: Boolean,
      default: false
    },
    title: {
      type: String,
      default: ''
    },
    size: {
      type: String,
      default: 'medium',
      validator: (value) => ['small', 'medium', 'large', 'fullscreen'].includes(value)
    },
    closable: {
      type: Boolean,
      default: true
    },
    closeOnOverlay: {
      type: Boolean,
      default: true
    },
    closeOnEsc: {
      type: Boolean,
      default: true
    }
  },
  
  emits: ['close'],
  
  setup(props, { emit }) {
    const close = () => {
      emit('close')
    }
    
    const handleOverlayClick = () => {
      if (props.closeOnOverlay) {
        close()
      }
    }
    
    const handleKeydown = (e) => {
      if (e.key === 'Escape' && props.closeOnEsc && props.isOpen) {
        close()
      }
    }
    
    // Lock body scroll when modal is open
    watch(() => props.isOpen, (isOpen) => {
      if (isOpen) {
        document.body.style.overflow = 'hidden'
      } else {
        document.body.style.overflow = ''
      }
    })
    
    onMounted(() => {
      window.addEventListener('keydown', handleKeydown)
    })
    
    onUnmounted(() => {
      window.removeEventListener('keydown', handleKeydown)
      document.body.style.overflow = ''
    })
    
    return {
      close,
      handleOverlayClick
    }
  }
}
</script>
